project(nlp)

if(USE_NLP)

#    set (CMAKE_C_STANDARD 99)

    set(SRCS
        nlp.hpp
        nlp.cpp
        )

    set(NLP_LIBRARIES
        prim
        kbm
        consh
        words
        lite
        odbc
        )

    include_directories(
        ./../../system/include
        ./../../rtl/eclrtl
        ./../../rtl/include
        ./../../common/deftype
        ./../../system/jlib

        ./nlp-engine/include
        ./nlp-engine/include/Api/lite
        ./nlp-engine/include/Api
        ./nlp-engine/cs/include
        /usr/local/libiodbc-3.52.12/include
        )

    add_definitions(-D_USRDLL -DECL_NLP_EXPORTS)

    HPCC_ADD_LIBRARY(nlp SHARED ${SRCS} ${UBER_NLP_SOURCE_FILES})
    if(${CMAKE_VERSION} VERSION_LESS "2.8.9")
        message(WARNING "Cannot set NO_SONAME. shlibdeps will give warnings when package is installed")
    elseif(NOT APPLE)
        set_target_properties(nlp PROPERTIES NO_SONAME 1)
    endif()

    target_compile_definitions(nlp PUBLIC NLP_HAVE_ALLOCA)

    install(
        TARGETS nlp
        DESTINATION plugins)

    target_link_libraries(nlp
        eclrtl
        jlib
        ${NLP_LIBRARIES})

endif(USE_NLP)

if(PLATFORM OR CLIENTTOOLS_ONLY)
    install(
        FILES ${CMAKE_CURRENT_SOURCE_DIR}/lib_nlp.ecllib
        DESTINATION plugins
        COMPONENT Runtime)
endif()
